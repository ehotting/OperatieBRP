select gio.onderzoek, naam, objectsleutelgegeven, lo3categorie, lo3stapelvolgnr, lo3voorkomenvolgnr, nt.indvoorkomentbvlevmuts 
from kern.gegeveninonderzoek gio join kern.element e on e.id = gio.element
join kern.his_persverblijfsr nt on gio.voorkomensleutelgegeven = nt.id
left join verconv.lo3voorkomen lo on lo.actie = nt.actieinh
where naam like 'Persoon.Verblijfsrecht%'
order by naam, onderzoek;
